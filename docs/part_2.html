dc_clean <- dc |> #cleans data collection
  filter(
    !is.na(code),
    !code %in% junk,
    !str_detect(code, "^(MOBA|FPS)$")) # only exact MOBA and FPS without numbers after

pc_clean <- pc |> #cleans participant contact
  filter(
    !is.na(id_participant_code),
    !id_participant_code %in% junk,
    !str_detect(id_participant_code, "^(MOBA|FPS)$"))

joined <- inner_join( #joins data as long as the match 
  pc_clean,
  dc_clean,
  by = c("id_participant_code" = "code"))

joined <- joined |>
  mutate(
    # Rename m_f to standardize to F / M
    sex = case_when(
      grepl("^f", m_f, ignore.case = TRUE) ~ "F",
      grepl("^m", m_f, ignore.case = TRUE) ~ "M",
      TRUE ~ NA_character_),
    # Standardize dominant_hand to R / L / Ambi
    dominant_hand = case_when(
      str_detect(dominant_hand, regex("^r", ignore_case = TRUE)) ~ "R",
      str_detect(dominant_hand, regex("^l", ignore_case = TRUE)) ~ "L",
      TRUE ~ NA_character_))|>
  select(
    -matches("^x\\d+$"),
    -any_of(drop_cols))|>
  filter(!is.na(dominant_hand), !is.na(sex))
